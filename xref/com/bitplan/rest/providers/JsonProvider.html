24  import java.io.InputStream;
25  import java.io.InputStreamReader;
26  import java.io.OutputStream;
27  import java.io.Reader;
28  import java.lang.annotation.Annotation;
29  import java.lang.reflect.ParameterizedType;
30  import java.lang.reflect.Type;
31  import java.util.ArrayList;
32  import java.util.List;
33  import java.util.Map;

35  import javax.ws.rs.Consumes;
36  import javax.ws.rs.Produces;
37  import javax.ws.rs.WebApplicationException;
38  import javax.ws.rs.core.Context;
39  import javax.ws.rs.core.MediaType;
40  import javax.ws.rs.core.MultivaluedMap;
41  import javax.ws.rs.ext.ContextResolver;
42  import javax.ws.rs.ext.MessageBodyReader;
43  import javax.ws.rs.ext.MessageBodyWriter;
44  import javax.ws.rs.ext.Provider;
45  import javax.ws.rs.ext.Providers;
46  import javax.xml.bind.JAXBContext;
47  import javax.xml.bind.JAXBException;
48  import javax.xml.bind.Marshaller;
49  import javax.xml.bind.Unmarshaller;
50  import javax.xml.transform.stream.StreamSource;

52  /**
53   * see http://blog.bdoughan.com/2012/03/moxy-as-your-jax-rs-json-provider.html
54   * @author wf
55   *
56   */
57  @Provider
58  @Produces(MediaType.APPLICATION_JSON)
59  @Consumes(MediaType.APPLICATION_JSON)
60  public class JsonProvider implements
61      MessageBodyReader<Object>, MessageBodyWriter<Object>{
62     
63      private static final String CHARSET = "charset";
64   
65      @Context
66      protected Providers providers;
67     
68      public static List<Class<?>> registeredTypes=new ArrayList<Class<?>>();
69      /**
70       * register a type
71       * @param clazz
72       */
73      public static void registerType(Class<?> clazz) {
74        registeredTypes.add(clazz);
75      }
76      
77      /**
78       * check whether the genericType is readable
79       */
80      public boolean isReadable(Class<?> type, Type genericType,
81          Annotation[] annotations, MediaType mediaType) {
82          boolean readable=registeredTypes.contains(getDomainClass(genericType));
83          return readable;
84      }
85     
86      /**
87       * read
88       */
89      public Object readFrom(Class<Object> type, Type genericType,
90              Annotation[] annotations, MediaType mediaType,
91              MultivaluedMap<String, String> httpHeaders, InputStream entityStream)
92              throws IOException, WebApplicationException {
93              try {
94                  Class<?> domainClass = getDomainClass(genericType);
95                  Unmarshaller u = getJAXBContext(domainClass, mediaType).createUnmarshaller();
96                  u.setProperty("eclipselink.media-type", MediaType.APPLICATION_JSON);
97                  u.setProperty("eclipselink.json.include-root", false);
98   
99                  StreamSource jsonSource;
100                 Map<String, String> mediaTypeParameters = mediaType.getParameters();
101                 if(mediaTypeParameters.containsKey(CHARSET)) {
102                     String charSet = mediaTypeParameters.get(CHARSET);
103                     Reader entityReader = new InputStreamReader(entityStream, charSet);
104                     jsonSource = new StreamSource(entityReader);
105                 } else {
106                     jsonSource = new StreamSource(entityStream);
107                 }
108  
109                 return u.unmarshal(jsonSource, domainClass).getValue();
110             } catch(JAXBException jaxbException) {
111                 throw new WebApplicationException(jaxbException);
112             }
113     }
114    
115     public boolean isWriteable(Class<?> type, Type genericType,
116         Annotation[] annotations, MediaType mediaType) {
117         return isReadable(type, genericType, annotations, mediaType);
118     }
119    
120     /**
121      * write
122      */
123     public void writeTo(Object object, Class<?> type, Type genericType,
124         Annotation[] annotations, MediaType mediaType,
125         MultivaluedMap<String, Object> httpHeaders,
126         OutputStream entityStream) throws IOException,
127         WebApplicationException {
128         try {
129             Class<?> domainClass = getDomainClass(genericType);
130             Marshaller m = getJAXBContext(domainClass, mediaType).createMarshaller();
131             m.setProperty("eclipselink.media-type", MediaType.APPLICATION_JSON);
132             m.setProperty("eclipselink.json.include-root", false);
133  
134             Map<String, String> mediaTypeParameters = mediaType.getParameters();
135             if(mediaTypeParameters.containsKey(CHARSET)) {
136                 String charSet = mediaTypeParameters.get(CHARSET);
137                 m.setProperty(Marshaller.JAXB_ENCODING, charSet);
138             }
139  
140             m.marshal(object, entityStream);
141         } catch(JAXBException jaxbException) {
142             throw new WebApplicationException(jaxbException);
143         }
144     }
145    
146     public long getSize(Object t, Class<?> type, Type genericType,
147         Annotation[] annotations, MediaType mediaType) {
148         return -1;
149     }
150    
151     private JAXBContext getJAXBContext(Class<?> type, MediaType mediaType)
152         throws JAXBException {
153         ContextResolver<JAXBContext> resolver
154             = providers.getContextResolver(JAXBContext.class, mediaType);
155         JAXBContext jaxbContext;
156         if(null == resolver || null == (jaxbContext = resolver.getContext(type))) {
157             return JAXBContext.newInstance(type);
158         } else {
159             return jaxbContext;
160         }
161     }
162    
163     private Class<?> getDomainClass(Type genericType) {
164         if(genericType instanceof Class) {
165             return (Class<?>) genericType;
166         } else if(genericType instanceof ParameterizedType) {
167             return (Class<?>) ((ParameterizedType) genericType).getActualTypeArguments()[0];
168         } else {
169             return null;
170         }
171     }
