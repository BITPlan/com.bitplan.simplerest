256   /**
257    * create a Server based on an url and possibly a ResourceConfig
258    * 
259    * @param url
260    * @param rc
261    * @param secure
262    *          - true if SSL should be used
263    * @param contextPath
264    * @return the HttpServer
265    * @throws Exception
266    */
267   public HttpServer createHttpServer(String url, ResourceConfig rc,
268       boolean secure, String contextPath) throws Exception {
269     // HttpServer result = GrizzlyServerFactory.createHttpServer(url, rc);
270     // http://grepcode.com/file/repo1.maven.org/maven2/com.sun.jersey/jersey-grizzly2/1.6/com/sun/jersey/api/container/grizzly2/GrizzlyServerFactory.java#GrizzlyServerFactory.createHttpServer%28java.net.URI%2Ccom.sun.jersey.api.container.grizzly2.ResourceConfig%29
271     HttpServer result = new HttpServer();
272     final NetworkListener listener = new NetworkListener("grizzly",
273         settings.getHost(), settings.getPort());
274     result.addListener(listener);
275     // do we need SSL?
276     if (secure) {
277       listener.setSecure(secure);
278       SSLEngineConfigurator sslEngineConfigurator = createSSLConfig(true);
279       listener.setSSLEngineConfig(sslEngineConfigurator);
280     }
281     // Map the path to the processor.
282     final ServerConfiguration config = result.getServerConfiguration();
283     final HttpHandler handler = ContainerFactory.createContainer(
284         HttpHandler.class, rc);
285     config.addHttpHandler(handler, contextPath);
286     return result;
287   }

292   /**
293    * get the given Store
294    * 
295    * @param type
296    * @param name
297    * @return the store file
298    * @throws IOException
299    */
300   public File getStoreFile(String type, String name) throws IOException {
301     final ClassLoader cl = RestServerImpl.class.getClassLoader();
302     URL surl = cl.getResource(name);
303     File result = null;
304     if (surl != null) {
305       LOGGER.log(Level.WARNING, "getting " + type + " from " + surl.toString());
306       result = File.createTempFile(name, ".tmp");
307       FileUtils.copyURLToFile(surl, result);
308     } else {
309       LOGGER.log(Level.WARNING, "could not get " + type + " from resource "
310           + name);
311     }
312     return result;
313   }

315   /**
316    * create SSL Configuration
317    * 
318    * @param isServer
319    *          true if this is for the server
320    * @return
321    * @throws Exception
322    */
323   private SSLEngineConfigurator createSSLConfig(boolean isServer)
324       throws Exception {
325     final SSLContextConfigurator sslContextConfigurator = new SSLContextConfigurator();
326     // override system properties
327     final File cacerts = getStoreFile("server truststore",
328         "truststore_server.jks");
329     if (cacerts != null) {
330       sslContextConfigurator.setTrustStoreFile(cacerts.getAbsolutePath());
331       sslContextConfigurator.setTrustStorePass(TRUSTSTORE_PASSWORD);
332     }
333 
334     // override system properties
335     final File keystore = getStoreFile("server keystore", "keystore_server.jks");
336     if (keystore != null) {
337       sslContextConfigurator.setKeyStoreFile(keystore.getAbsolutePath());
338       sslContextConfigurator.setKeyStorePass(TRUSTSTORE_PASSWORD);
339     }
340 
341     //
342     boolean clientMode = false;
343     // force client Authentication ...
344     boolean needClientAuth = settings.isNeedClientAuth();
345     boolean wantClientAuth = settings.isWantClientAuth();
346     SSLEngineConfigurator result = new SSLEngineConfigurator(
347         sslContextConfigurator.createSSLContext(), clientMode, needClientAuth,
348         wantClientAuth);
349     return result;
350   }

373     }
374     for (String headerName : req.getHeaderNames()) {
375       System.out.println("req header: " + headerName + "="
376           + req.getHeader(headerName));

380   /**
381    * get the SSL Client certificate for the given request (if any)
382    * 
383    * @param req
384    *          - the request to check
385    * @return - the X509Certificate
386    */
387   protected X509Certificate getCertificate(Request req) {
388     X509Certificate result = null;
389     Object certobj = req.getAttribute(Globals.CERTIFICATES_ATTR);
390     if (certobj != null) {
391       String msg = "certificate " + certobj.getClass().getName() + " found";
392       LOGGER.finer(msg);
393       if (certobj instanceof java.security.cert.X509Certificate[]) {
394         java.security.cert.X509Certificate[] certs = (X509Certificate[]) certobj;
395         if (certs.length > 0)
396           result = certs[0];
397       }
398     }
399     return result;
400   }

416   /**
417    * check that the SSL Client Certificate is valid and return the checked
418    * Principal
419    * 
420    * @param req
421    */
422   protected Principal checkSSLClientCertificate(Request req) {
423     X509Certificate clientCert = this.getCertificate(req);
424     Principal result = null;
425     if (clientCert != null) {
426       Principal subjectDN = clientCert.getSubjectDN();
427       try {
428         result = checkPrincipal(subjectDN);
429       } catch (Exception e) {
430         LOGGER.log(Level.SEVERE,
431             "checkPrincipal failed with exception " + e.getMessage());
432       }
433     } else {
434       LOGGER.log(Level.SEVERE,
435           "SSL Client certificate is missing for " + req.getRequestURI());
436     }
437     return result;
438   }

448   public void addHttpHandler(String context, String path, boolean staticHandler) {
449     HttpHandler handler = null;
450     String type = "";
451     if (staticHandler) {
452       handler = new StaticHttpHandler(context);
453       type = "static";
454     } else {
455       handler = new CLStaticHttpHandler(this.getClass().getClassLoader(),
456           context);
457       type = "classpath";
458     }
459     if (!path.endsWith("/")) {
460       path += "/";
461     }
462     LOGGER.log(Level.INFO, "adding " + type + " httphandler " + context + "->"
463         + path);
464     httpServer.getServerConfiguration().addHttpHandler(handler, path);
465   }

467   @SuppressWarnings("unchecked")
468   @Override
469   public synchronized void createServer() throws Exception {
470     if (settings == null)
471       throw new IllegalArgumentException(
472           "Can't start RestServer Settings are not set");
473     // / create a Grizzly Server
474     String url = getUrl();
475     // http://jersey.java.net/nonav/documentation/latest/user-guide.html#d4e52
476 
477     // which packages do ew need to configure?
478     String packages = settings.getPackages();
479     context = null;
480     try {
481       if (packages!=null) {
482         // do we want basic authentication with a user manager?
483         if (settings.getUserManager()!=null) {
484           // add the package with the BasicAuthSecurityProvider
485           packages+="com.bitplan.rest.basicauth;"+packages;
486         }
487         String pa[] = packages.split(";");
488         ResourceConfig rc = new PackagesResourceConfig(pa);
489         // more config for provider
490         UserManager userManager = settings.getUserManager();
491         if (userManager!=null) {
492           rc.getContainerRequestFilters().add(new BasicAuthSecurityProvider(userManager));
493         }
494         // http://stackoverflow.com/questions/3677064/jax-rs-jersey-howto-force-a-response-contenttype-overwrite-content-negotiatio
495         rc.getMediaTypeMappings().put("json", MediaType.APPLICATION_JSON_TYPE);
496         rc.getMediaTypeMappings().put("xml", MediaType.APPLICATION_XML_TYPE);
497         rc.getMediaTypeMappings().put("txt", MediaType.TEXT_PLAIN_TYPE);
498         // FIXME
499         rc.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING,
500             Boolean.TRUE);
501 
502         String[] containerRequestFilters = settings
503             .getContainerRequestFilters();
504         if (containerRequestFilters != null) {
505           rc.getProperties().put(
506               ResourceConfig.PROPERTY_CONTAINER_REQUEST_FILTERS,
507               containerRequestFilters);
508         }
509         if (useServlet) {
510           // http://stackoverflow.com/questions/15769415/grizzly-and-servletcontainercontext
511           httpServer = GrizzlyServerFactory.createHttpServer(new URI(url),
512               new HttpHandler() {
513 
514                 @Override
515                 public void service(Request rqst, Response rspns)
516                     throws Exception {
517                   rspns.setStatus(404, "Not found");
518                   rspns.getWriter().write("404: not found");
519                 }
520               });
521           context = new WebappContext("Context", "");
522           ServletRegistration registration = context.addServlet(
523               "ServletContainer", ServletContainer.class);
524           registration.setInitParameter(
525               "com.sun.jersey.config.property.packages", packages);
526           // add security filter (which handles http basic authentication)
527           /**
528            * registration.setInitParameter(
529            * ResourceConfig.PROPERTY_CONTAINER_REQUEST_FILTERS,
530            * containerRequestFilters);
531            */
532           registration.addMapping("/*");
533         } else {
534           httpServer = createHttpServer(url, rc, settings.isSecure(),
535               settings.getContextPath());
536         }
537       } else {
538         httpServer = createHttpServer(url, null, settings.isSecure(),
539             settings.getContextPath());
540       }
541     } catch (Throwable th) {
542       System.err.println(th.getMessage());
543       th.printStackTrace();
544     }
545 
546     if (httpServer == null) {
547       throw new Exception("Couldn't create http server");
548     }
549 
550     // remote address handling
551     httpServer.getServerConfiguration().getMonitoringConfig()
552         .getWebServerConfig().addProbes(new HttpServerProbe.Adapter() {
553           @SuppressWarnings("rawtypes")
554           @Override
555           public void onRequestReceiveEvent(HttpServerFilter filter,
556               Connection connection, Request request) {
557             String remote_addr = request.getRemoteAddr();
558             HttpRequestPacket req = request.getRequest();
559             String msg = "remote addr is " + remote_addr
560                 + " for request of type " + req.getClass().getName();
561             if (settings.isDebug())
562               LOGGER.info(msg);
563             req.setHeader("remote_addr", remote_addr);
564             // showDebug(request);
565             Principal principal=null;
566             if (settings.isSecure()) {
567               principal = RestServerImpl.this
568                   .checkSSLClientCertificate(request);
569               if (principal == null) {
570                 // FIXME - we'd like to get a proper status here!
571                 if (settings.isNeedClientAuth()) {
572                   LOGGER.log(Level.SEVERE,
573                       "no principal with needClientAuth=true");
574                   request.setRequestURI("invalidaccess");
575                 }
576               }
577             } 
578             // did we get a principal for this request?
579             if (principal != null) {
580               // http://stackoverflow.com/questions/909185/jersey-security-and-session-management
581               // LOGGER.info("remote user: "+request.getRemoteUser());
582 
583               // this would have been simple but does not work see
584               // http://stackoverflow.com/questions/20105005/unsupportedoperationexception-getuserprincipal
585               request.setUserPrincipal(principal);
586 
587               // work-around
588               PrincipalCache.add(principal);
589               String principal_id = PrincipalCache.getId(principal);
590               req.setHeader("principal_id", principal_id);
591             }
592             // debug(req);
593 
594           }
595 
596           @SuppressWarnings("rawtypes")
597           @Override
598           public void onRequestCompleteEvent(HttpServerFilter filter,
599               Connection connection, Response response) {
600             // do nothing
601             if (settings.isDebug())
602               LOGGER.info("request complete");
603           }
604         });
605 
606     // add configured classpath handlers
607     for (String path : this.getSettings().getClassPathHandlers().keySet()) {
608       String clPath = this.getSettings().getClassPathHandlers().get(path);
609       // relative or absolute path?
610       if (!path.startsWith("/")) {
611         path = settings.getContextPath() + "/" + path;
612       }
613       addHttpHandler(clPath, path, false);
614     }
615 
616     // set default encoding
617     httpServer.getServerConfiguration().setDefaultQueryEncoding(
618         Charsets.UTF8_CHARSET);
619   }

635     if (auth==null) {
636       throw new WebApplicationException(Status.UNAUTHORIZED);

658   @Override
659   public synchronized void startWebServer() throws Exception {
660     createServer();
661     // start server
662     httpServer.start();
663     if (context != null) {
664       context.deploy(httpServer);
665     }
666 
667     // FIXME?
668     /*
669      * if (guiceModule != null) { WebappContext context =
670      * createWebappContext(""); context.deploy(srv); }
671      */
672     LOGGER.log(Level.INFO, "starting server for URL: " + getUrl());
673     running = true;
674     informStarter();
675     int sleep = 0;
676     // timeOut is in Secs - we go by 1/20th of sec = 50 millisecs
677     while (running && (sleep < settings.getTimeOut() * 20)) {
678       // System.out.println(sleep+":"+running);
679       // sleep 50 millisecs
680       Thread.sleep(50);
681       sleep++;
682     }
683     stop();
684   }
