21  package com.bitplan.rest.users;
22  
23  import java.util.ArrayList;
24  import java.util.HashMap;
25  import java.util.List;
26  import java.util.Map;
27  
28  import javax.xml.bind.annotation.XmlElement;
29  import javax.xml.bind.annotation.XmlElementWrapper;
30  import javax.xml.bind.annotation.XmlRootElement;
31  import javax.xml.bind.annotation.XmlTransient;
32  
33  import com.bitplan.jaxb.JaxbFactory;
34  import com.bitplan.jaxb.JaxbFactoryApi;
35  import com.bitplan.jaxb.ManagerImpl;
36  import com.bitplan.rest.Crypt;
37  import com.bitplan.rest.CryptImpl;
38  import com.bitplan.rest.User;
39  import com.bitplan.rest.UserManager;
40  
41  @XmlRootElement(name="UserManager")
42  public class UserManagerImpl extends ManagerImpl<UserManagerImpl,User> implements UserManager {
43    static JaxbFactory<UserManagerImpl> jaxbFactory;
44    
45    transient List<User> users=new ArrayList<User>();
46    @XmlTransient
47    Map<String,User> userById=new HashMap<String,User>();
48  
49    @XmlTransient
50    private CryptImpl crypt;
51  
52    /**
53     * @return the users
54     */
55    @XmlElementWrapper(name="users")
56    @XmlElement(name="User", type=UserImpl.class)
57    public List<User> getUsers() {
58      return users;
59    }
60  
61    /**
62     * @param users the users to set
63     */
64    public void setUsers(List<User> users) {
65      this.users = users;
66    }
67    
68    /**
69     * get the JAXBFactory for this class
70     * @return the factory
71     */
72    public static JaxbFactory<UserManagerImpl> getJaxbFactory() {
73      if (jaxbFactory==null) {
74        jaxbFactory=new JaxbFactory<UserManagerImpl>(UserManagerImpl.class,ObjectFactory.class);
75      }
76      return jaxbFactory;
77    }
78    
79    @Override
80    public JaxbFactoryApi<UserManagerImpl> getFactory() {
81      return getJaxbFactory();
82    }
83   
84  
85    @Override
86    public void add(User user) {
87      this.users.add(user);
88      this.userById.put(user.getId(), user);
89    }
90    
91    /**
92     * get a user by the given id
93     * @param id - the user id to lookup by
94     * @return the user
95     */
96    public User getById(String id) {
97      User user=this.userById.get(id);
98      return user;
99    }
100 
101   @Override
102   public Crypt getCrypt() {
103     if (crypt==null) {
104       crypt=new CryptImpl("YzMhYb57ljt4pR3rbklA3w8V1NWdojRa","s5qzAZ9x");
105     }
106     return crypt;
107   }
108   
109   /**
110    * reinitialize the UserById Map
111    */
112   public void reinitUserById() {
113     List<User> lusers = getUsers();
114     for (User luser:lusers) {
115       userById.put(luser.getId(), luser);
116     }    
117   }
118 
119   /**
120    * create a user manager form the given xml string
121    * @param xml - the xml string
122    * @return the user manager
123    * @throws Exception
124    */
125   public static UserManager fromXml(String xml) throws Exception {
126     UserManagerImpl um=(UserManagerImpl) getJaxbFactory().fromXML(xml);
127     um.reinitUserById();
128     return um;
129   }
130   
131   private static UserManagerImpl instance=null;
132   
133   /**
134    * get a UserManager Instance
135    * @return the instance
136    */
137   public static UserManagerImpl getInstance() {
138     if (instance==null) {
139       instance=new UserManagerImpl();
140     }
141     return instance;
142   }
143 
144   @Override
145   @XmlTransient
146   public List<User> getElements() {
147     return this.getUsers();
148   }
149 
150 }
