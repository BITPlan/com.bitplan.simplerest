37  public class TripleQueryImpl implements TripleQuery {
38    public static enum TopicType {
39      subject, predicate, object
40    };
41  
42    TripleStore tripleStore;
43    public Triple triple;
44    TopicType topicType;
45    protected  List<Triple> triples;
46  
47    /**
48     * create a TripleQuery from the given tripleStore
49     * @param tripleStore
50     */
51    public TripleQueryImpl(TripleStore tripleStore) {
52      this.tripleStore=tripleStore;
53      this.topicType=TopicType.subject;
54      setTriples(tripleStore.getTriples());
55    }
56    
57    /**
58     * create a new TripleQuery based on an existing one
59     * @param tripleQuery
60     */
61    public TripleQueryImpl(TripleQueryImpl tripleQuery) {
62      this.tripleStore=tripleQuery.tripleStore;
63      this.topicType=tripleQuery.topicType;
64      this.triples=tripleQuery.triples;
65      this.triple=tripleQuery.triple;
66    }
67  
68    /**
69     * change the query topic
70     * @param topicType
71     * @return the query
72     */
73    public TripleQueryImpl queryTopic(TopicType topicType) {
74      TripleQueryImpl result = new TripleQueryImpl(this);
75      result.topicType=topicType; 
76      return result;
77    }
78    
79    /**
80     * get the TripleContainer
81     * @return the tripleContainer of the current topic type
82     */
83    public TripleContainer getTripleContainer() {
84      TripleContainer result;
85      switch (topicType) {
86      case subject:
87        result = tripleStore.bySubject;
88        break;
89      case predicate:
90        result = tripleStore.byPredicate;
91        break;
92      case object:
93        result = tripleStore.bySubject;
94        break;
95      default:
96        throw new IllegalArgumentException("unknown topic Type: "+topicType);
97      }
98      return result;
99    }
100   
101   /**
102    * get the Topic for the current topic type subject predicate or object
103    * @return the object
104    */
105   public Object getTopic() {
106     Object topic=null;
107     if (triple!=null) {
108       switch (topicType) {
109       case subject:
110         topic=triple.getSubject();
111         break;
112       case predicate:
113         topic=triple.getPredicate();
114         break;
115       case object:
116         topic=triple.getObject();
117         break;
118       }
119     }
120     return topic;
121   }

126   @Override
127   public List<Triple> select(Object subject, Object predicate, Object object) {
128     TripleQueryImpl tripleQuery = this.query(subject,
129         predicate, object);
130     List<Triple> result = tripleQuery.triples;
131     return result;
132   }

134   /**
135    * select the triples for the given "template" triple
136    * 
137    * @param select
138    * @return the selected list of triples
139    */
140   public List<Triple> select(Triple select) {
141     TripleQueryImpl tripleQuery = this.queryTriple(select);
142     List<Triple> result = tripleQuery.triples;
143     return result;
144   }

146   /* (non-Javadoc)
147    * @see org.sidif.triple.TripleQueryI#same(java.lang.Object, java.lang.Object)
148    */
149   @Override
150   public boolean same(Object object, Object other) {
151     boolean result = false;
152     if ((object == null) && (other == null))
153       result = true;
154     if ((object != null) && (other != null))
155       result = object.equals(other);
156     return result;
157   }

159   /* (non-Javadoc)
160    * @see org.sidif.triple.TripleQueryI#query(java.lang.Object, java.lang.Object, java.lang.Object)
161    */
162   @Override
163   public TripleQueryImpl query(Object subject,
164       Object predicate, Object object) {
165     TripleImpl select = new TripleImpl(subject, predicate, object);
166     TripleQueryImpl result = queryTriple(select);
167     return result;
168   }

170   /* (non-Javadoc)
171    * @see org.sidif.triple.TripleQueryI#selectSingle(java.lang.Object, java.lang.Object, java.lang.Object)
172    */
173   @Override
174   public Triple selectSingle(Object subject, Object predicate, Object object) {
175     List<Triple> triples = this.select(subject, predicate, object);
176     if (triples.size() > 1) {
177       String msg="selectSingle subject='"+subject
178           +"' predicate='"+predicate
179           +"' object='"+object+"' returned "+triples.size()+" triples but there should be only one!";
180       // we'll show the first (upto) 10 culprits
181       for (int i=1;i<=Math.min(triples.size(), 10);i++) {
182         Triple triple=triples.get(i-1);
183         msg+="\n\t"+i+":"+triple.toString();
184       }
185       throw new IllegalStateException(msg);
186     } else if (triples.size()==0) {
187       return null;
188     }
189     return triples.get(0);
190   }

192   /* (non-Javadoc)
193    * @see org.sidif.triple.TripleQueryI#queryTriple(org.sidif.triple.Triple)
194    */
195   @Override
196   public TripleQueryImpl queryTriple(Triple select) {
197     TripleQueryImpl result=new TripleQueryImpl(this);
198     result.triples = new ArrayList<Triple>();
199     // FIXME - this is inefficient
200     for (Triple triple : this.getTriples()) {
201       boolean subjectOk = select.getSubject() == null
202           || same(triple.getSubject(), select.getSubject());
203       boolean predicateOk = select.getPredicate() == null
204           || same(triple.getPredicate(), select.getPredicate());
205       boolean objectOk = select.getObject() == null
206           || same(triple.getObject(), select.getObject());
207       if (subjectOk && predicateOk && objectOk) {
208         result.add(triple);
209       }
210     }
211     result.setTriple();
212     return result;
213   }
214   
215   /**
216    * add the given Triple to my triples
217    * @param triple
218    */
219   private void add(Triple triple) {
220     if (!triples.contains(triple)) {
221       triples.add(triple);
222     }
223   }

225   /**
226    * set my triples
227    * @param pTriples - the triples to use
228    */
229   public void setTriples(List<Triple> pTriples) {
230     triples=pTriples;
231     setTriple();
232   }
233   
234   /**
235    * set my triple (head triple pointer ...)
236    */
237   public void setTriple() {
238     if (triples.size()>0)
239       triple=triples.get(0);
240     else
241       triple=null;  
242   }

244   /* (non-Javadoc)
245    * @see org.sidif.triple.TripleQueryI#union(org.sidif.triple.TripleQueryImpl)
246    */
247   @Override
248   public TripleQuery union(TripleQuery other) {
249     TripleQueryImpl result=new TripleQueryImpl(this);
250     List<Triple> triples = new ArrayList<Triple>();
251     triples.addAll(this.getTriples());
252     triples.addAll(other.getTriples());
253     result.setTriples(triples);
254     return result;
255   }

257   /* (non-Javadoc)
258    * @see org.sidif.triple.TripleQueryI#intersect(org.sidif.triple.TripleQueryI)
259    */
260   @Override
261   public TripleQuery intersect(TripleQuery otherQuery) {
262     TripleQueryImpl result=new TripleQueryImpl(this);
263     result.triples=new ArrayList<Triple>();
264     // FIXME this is ineffecient quadratic O(this.size x otherQuery.size) ...
265     for (Triple triple:this.getTriples()) {
266       for (Triple otherTriple:otherQuery.getTriples()) {
267         if (same(triple,otherTriple)) {
268           result.add(triple);
269         }
270       }
271     }
272     result.setTriple();
273     return result;
274   }

276   /* (non-Javadoc)
277    * @see org.sidif.triple.TripleQueryI#complement(org.sidif.triple.TripleQueryI)
278    */
279   @Override
280   public TripleQuery complement(TripleQuery otherQuery) {
281     TripleQueryImpl result=new TripleQueryImpl(this);
282     result.triples=new ArrayList<Triple>();
283     // FIXME this is ineffecient quadratic O(this.size x otherQuery.size) ...
284     for (Triple triple:this.getTriples()) {
285       for (Triple otherTriple:otherQuery.getTriples()) {
286         if (!same(triple,otherTriple)) {
287           result.add(triple);
288         }
289       }
290     }
291     result.setTriple();
292     return result;
293   }
