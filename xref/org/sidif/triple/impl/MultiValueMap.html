42  import java.util.List; 
43  import java.util.Map; 
44   
45  /**
46   * Extension of the {@code Map} interface that stores multiple values. 
47   * 
48   * @author Arjen Poutsma 
49   * @since 3.0 
50   */ 
51  public interface MultiValueMap<K, V> extends Map<K, List<V>> { 
52   
53   /**
54    * Return the first value for the given key. 
55    * @param key the key 
56    * @return the first value for the specified key, or <code>null</code> 
57    */ 
58   V getFirst(K key); 
59   
60   /**
61    * Add the given single value to the current list of values for the given key. 
62    * @param key the key 
63    * @param value the value to be added 
64    */ 
65   void add(K key, V value); 
66   
67   /**
68    * Set the given single value under the given key. 
69    * @param key the key 
70    * @param value the value to set 
71    */ 
72   void set(K key, V value); 
73   
74   /**
75    * Set the given values under. 
76    * @param values the values. 
77    */ 
78   void setAll(Map<K, V> values); 
79   
80   /**
81    * Returns the first values contained in this {@code MultiValueMap}. 
82    * @return a single value representation of this map 
83    */ 
84   Map<K, V> toSingleValueMap(); 
85   
86  }
