21  package org.sidif.util;
22  
23  import org.sidif.triple.Triple;
24  import org.sidif.triple.TripleStore;
25  import org.sidif.triple.Value;
26  
27  /**
28   * creates SiDIF Output for a given TripleStore
29   * 
30   * @author wf
31   *
32   */
33  public class SiDIFWriter {
34    private static Object currentSubject;
35    private static boolean canonical;
36  
37    /**
38     * get a sidif String from the given TripleStore
39     * @param tripleStore
40     * @return the sidif string created for the given TripleStore
41     */
42    public static String asSiDIF(TripleStore tripleStore) {
43      String result=asSiDIF(tripleStore,false);
44      return result;
45    }
46    
47    /**
48     * get the SiDIF representation of the given TripleStore
49     * 
50     * @param tripleStore
51     * @param canonical - if true the canonical version is used
52     * @return the sidif String 
53     */
54    public static String asSiDIF(TripleStore tripleStore, boolean canonical) {
55      SiDIFWriter.canonical=canonical;
56      String sidif = "";
57      currentSubject=null;
58      for (Triple triple : tripleStore.getTriples()) {
59        sidif += asSiDIF(triple);
60      }
61      return sidif;
62    }
63  
64    /**
65     * get the subject's to string or "it" if the subject is the current Subject
66     * @param subject
67     * @return the string
68     */
69    public static String getIt(Object subject) {
70      String subjectString;
71      if (!canonical && currentSubject!=null && subject.toString().equals(currentSubject.toString())) {
72        subjectString="it";
73      } else {
74        subjectString=subject.toString();
75      }
76      return subjectString;
77    }
78    
79    /**
80     * as SiDIF
81     * 
82     * @param triple
83     * @return the sidif representation of the triple
84     */
85    public static String asSiDIF(Triple triple) {
86      String sidif = "";
87      Object subject = triple.getSubject();
88      Object predicate = triple.getPredicate();
89      Object object = triple.getObject();
90      if (object instanceof Value) {
91        @SuppressWarnings("rawtypes")
92        Value value = (Value) object;  
93        if (value.type!=null && value.type.equals("Reference")) {
94          sidif += subject + " " + predicate + " " + getIt(value.getObject()) + "\n";       
95        } else {
96          sidif += value.asSiDIF()+ " is " +predicate  +" of "+getIt(subject) +"\n";
97        }
98      } else {
99        sidif += subject + " " + predicate + " " + object + "\n";
100     }
101     currentSubject=subject;
102     return sidif;
103   }
104 }
