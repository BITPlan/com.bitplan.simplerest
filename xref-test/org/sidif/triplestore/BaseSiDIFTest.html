21  package org.sidif.triplestore;
22  
23  import static org.junit.Assert.assertTrue;
24  
25  import java.io.File;
26  
27  import org.sidif.parser.jjtree.SiDIF;
28  import org.sidif.triple.TripleStore;
29  import org.sidif.util.SiDIFReader;
30  import org.sidif.util.TripleStoreBuilder;
31  
32  /**
33   * Base class for SiDIF tests
34   * 
35   * @author wf
36   *
37   */
38  public class BaseSiDIFTest {
39    /**
40     * Logging may be enabled by setting debug to true
41     */
42    protected static java.util.logging.Logger LOGGER = java.util.logging.Logger
43        .getLogger("org.sidif.triplestore");
44    
45    // set to true for debugging
46    private static boolean debug = false;
47    
48    public static boolean isDebug() {
49      return debug;
50    }
51  
52    public static void setDebug(boolean pDebug) {
53      debug = pDebug;
54    }
55  
56    /**
57     * get all example Names
58     * @return all example names
59     */
60    String[] getAllExampleNames() {
61      String examples[] = { "example1", "example2","familyTree","graph1", "json_ld_manu_sporny","notation3_TonyBenn",
62          "presentation", "rdf_cd", "rdf_json_anna_wilder", "royal92-14", "royal92", "trig_bob_alice", "turtle_spiderman","typetest",
63          "utf8","vcard" };
64      return examples;
65    }
66    
67    /**
68     * get the example file for the given exampleName
69     * 
70     * @param exampleName
71     * @return
72     */
73    public File getExampleFile(String exampleName) {
74      File exampleFile = new File("src/test/resources/sidif/" + exampleName
75          + ".sidif");
76      assertTrue(exampleName+" should exist",exampleFile.exists());
77      if (isDebug()) {
78        System.out.println("reading " + exampleFile.getName());
79      }
80      return exampleFile;
81    }
82  
83    /**
84     * get a SiDIF graph from the given example
85     * 
86     * @param exampleName
87     * @return the SiDIF parser result
88     * @throws Exception
89     */
90    public SiDIF getSiDIFFromExample(String exampleName) throws Exception {
91      File exampleFile = getExampleFile(exampleName);
92      SiDIF sidif = SiDIF.fromFile(exampleFile);
93      return sidif;
94    }
95  
96    /**
97     * get the TripleStore for the given example Name
98     * 
99     * @param exampleName
100    * @throws Exception
101    * @return the triple for the given example name
102    */
103   public TripleStore getTripleStoreFromExample(String exampleName)
104       throws Exception {
105     File exampleFile = getExampleFile(exampleName);
106     SiDIFReader siDIFReader = TripleStoreBuilder.getSiDIFReader();
107     TripleStore tripleStore=TripleStore.fromSiDIFFile(siDIFReader,exampleFile);
108     return tripleStore;
109   }

111 }
