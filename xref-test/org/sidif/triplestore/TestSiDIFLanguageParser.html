34  /**
35   * Unit tests for testing ANTLR based SiDIF Language Parser
36   * 
37   * @author wf
38   *
39   */
40  public class TestSiDIFLanguageParser extends BaseTest {
41  
42    @Override
43    public LanguageParser getParser() {
44      return new SiDIFLanguageParser();
45    }
46  
47    @Test
48    /**
49     * test a few simple SiDIF statements to be parseable
50     * @throws Exception
51     */
52    public void testSimpleSiDIF() throws Exception {
53      String sidifs[] = { "dad loves mum",
54          "true is value of booleanLiteral",
55          "0x7F is value of hexLiteral",
56          "1 is value of integer",
57          "'a' is value of char",
58          "\"a multi\nline\n-comment \n\" is value of comment",
59          "#\n#This is just acomment\n#over multiple lines\n1 is month of date",
60          "2.7 is floatvalue of it",
61          "2015-02-25 is exampleValue of Date\n" + 
62          "2015-02-25 11:34 is exampleValue of DateTime\n" + 
63          "1 is exampleValue of Number\n" + 
64          "3.1415926 is exampleValue of Number\n" + 
65          "\"Hello World!\" is exampleValue of Text\n" + 
66          "11:24 is exampleValue of Time\n" + 
67          "http://www.bitplan.com is exampleValue of IRI",
68          "https://semantic-mediawiki.org/wiki/Help:List_of_datatypes is example of IRI"
69       };
70      super.debug=BaseSiDIFTest.isDebug();
71      this.checkgui=true;
72      for (String sidif : sidifs) {
73        LanguageParser lparser = getParser();
74        lparser.setGui(true);
75        super.runParser(lparser, sidif, 0);
76      }
77    }
78  
79    @Test
80    /**
81     * test parse ability of SiDIF files in examples directory
82     */
83    public void testSiDIFSamples() throws Exception {
84      super.debug=BaseSiDIFTest.isDebug();
85      File rootDir = new File("src/test/resources");
86      String[] extensions = { ".sidif" };
87      checkgui = false;
88      super.checkTree=false;
89      super.gui=false;
90      List<SourceDirectory> sourceDirectories = new ArrayList<SourceDirectory>();
91      SourceDirectory sidifSrc = new SourceDirectory(
92          rootDir.getCanonicalPath() + "/sidif", "utf-8", "SiDIF samples");
93      sourceDirectories.add(sidifSrc);
94      int progressStep = 1;
95      int limit = 200;
96      String[] ignorePrefixes = {};
97      testParseFilesInDirectories(rootDir, sourceDirectories, extensions,
98          ignorePrefixes, limit, progressStep);
99    }
