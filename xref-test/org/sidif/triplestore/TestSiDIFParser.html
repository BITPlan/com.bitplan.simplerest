49    @Test
50    public void testSiDIFDump() throws ParseException {
51      String sidif1 = "Tokyo isA City\n"
52          + "\"http://www.metro.tokyo.jp/ENGLISH/ \" is webpage of Tokyo\n"
53          + "City isA Concept\n" + "Country isA Concept\n"
54          + "City linksTo Country\n" + "name addsTo City\n"
55          + "name isA Property\n" + "Property isA Concept\n"
56          + "Property linksTo Concept\n" + "Concept sameAs Topic";
57      StringReader reader = new StringReader(sidif1);
58      SiDIF sidif = new SiDIF(reader);
59      SiDIFVisitor visitor = new SiDIFDumpVisitor();
60      String dump = sidif.visit(visitor, "");
61      if (isDebug())
62        System.out.println(dump);
63    }

65    @Test
66    public void testSiDIFExamples() throws Exception {
67      final int maxLength = 10000; // FIXME performance issue for dump
68      for (String example : super.getAllExampleNames()) {
69        File exampleFile = this.getExampleFile(example);
70        if (exampleFile.length() < maxLength) {
71          SiDIF sidif = getSiDIFFromExample(example);
72          SiDIFVisitor visitor = new SiDIFDumpVisitor();
73          String dump = sidif.visit(visitor, "");
74          File dumpFile = new File(exampleFile.getPath()
75              .replace("sidif", "dump"));
76          FileUtils.writeStringToFile(dumpFile, dump);
77          String expected = FileUtils.readFileToString(dumpFile);
78          if (isDebug() && dump.length() < maxLength) {
79            System.out.println(dump);
80          }
81          assertEquals(example, expected, dump);
82        }
83      }
84    }

86    @Test
87    public void testTripleStore() throws Exception {
88      /**
89       * example1 (10)
90       * example2 (15)
91       * familyTree (51)
92       * graph1 (7)
93       * json_ld_manu_sporny (16)
94       * notation3_TonyBenn (4)
95       * presentation (39)
96       * rdf_cd (22)
97       * rdf_json_anna_wilder (8)
98       * royal92-14 (209)
99       * royal92 (25614)
100      * trig_bob_alice (11)
101      * turtle_spiderman (8)
102      * typetest (62)
103      * utf8 (3)
104      * vcard (31)
105      */
106     int expectedSize[] = { 11, 15, 51, 7, 16, 4, 545, 22, 8, 209, 25614, 11, 8,
107         62, 3, 31 };
108     int index = 0;
109     for (String example : getAllExampleNames()) {
110       TripleStore tripleStore = getTripleStoreFromExample(example);
111       int storeSize = tripleStore.size();
112       if (isDebug()) {
113         System.out.println(example + " (" + storeSize + ")");
114         TripleStoreDumper.dump(tripleStore);
115       }
116       assertEquals(example, expectedSize[index], storeSize);
117       index++;
118     }
119   }

121   @Test
122   public void testSelect() throws Exception {
123     TripleStore tripleStore = getTripleStoreFromExample("vcard");
124     if (isDebug())
125       TripleStoreDumper.dump(tripleStore);
126     TripleQuery cardTriples = tripleStore.query().query(null, "isA", "VCard");
127     // TripleContainer cardTriples = select(tripleStore,select);
128     int cardSize = cardTriples.size();
129     assertEquals(3, cardSize);
130     for (Triple cardTriple : cardTriples.getTriples()) {
131       if (isDebug())
132         TripleStoreDumper.dump(cardTriple, "card: ");
133       TripleImpl propertySelect = new TripleImpl(cardTriple.getSubject(), null, null);
134       TripleQuery cardProperties = tripleStore.query().queryTriple(propertySelect);
135       for (Triple cardProperty : cardProperties.getTriples()) {
136         if (isDebug())
137           TripleStoreDumper.dump(cardProperty, "");
138       }
139     }
