25  import org.junit.Test;

32  /**
33   * type test
34   * 
35   * @author wf
36   *
37   */
38  public class TestTypes extends BaseSiDIFTest {
39    @Test
40    public void testTypes() throws Exception {
41      TripleStore tripleStore = getTripleStoreFromExample("typetest");
42      // debug=true;
43      if (isDebug())
44        TripleStoreDumper.dump(tripleStore);
45      int tripleStoreSize = tripleStore.size();
46      assertEquals(62, tripleStoreSize);
47      int literals = 0;
48      for (Triple triple : tripleStore.getTriples()) {
49        if (isDebug())
50          TripleStoreDumper.dump(triple, "");
51        Object object = triple.getObject();
52        if (object != null) {
53          assertTrue(object instanceof Value);
54          @SuppressWarnings("unchecked")
55          Value<Object> value = (Value<Object>) object;
56          if (value.literal)
57            literals++;
58        }
59      }
60      if (isDebug())
61        System.out.println("found " + literals + " literals");
62      assertEquals(16, literals);
63    }
