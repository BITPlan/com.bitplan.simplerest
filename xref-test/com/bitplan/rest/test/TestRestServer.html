98    /**
99     * start the given Rest Server
100    * 
101    * @param rs
102    * @return
103    * @throws Exception
104    */
105   public void startServer(RestServer rs, boolean scanForAvailablePort)
106       throws Exception {
107     // find an available port
108     if (scanForAvailablePort) {
109       while (!rs.isPortAvailable(testPort))
110         testPort++;
111     }
112     rs.getSettings().setPort(testPort);
113     rs.setStarter(semaphore);
114     String args[] = {};
115     rs.startServer(args);
116     synchronized (semaphore) {
117       semaphore.wait();
118     }
119   }

179   /**
180    * upload the given file
181    * 
182    * inspired by
183    * http://neopatel.blogspot.de/2011/04/jersey-posting-multipart-data.html
184    * 
185    * @param url
186    * @param uploadFile
187    * @return the result
188    * @throws Exception 
189    */
190   public String upload(String url, File uploadFile) throws Exception {
191     WebResource resource = getResource(url);
192     FormDataMultiPart form = new FormDataMultiPart();
193     form.field("fileName", uploadFile.getName());
194     FormDataBodyPart fdp = new FormDataBodyPart("content", new FileInputStream(
195         uploadFile), MediaType.APPLICATION_OCTET_STREAM_TYPE);
196     form.bodyPart(fdp);
197     String response = resource.type(MediaType.MULTIPART_FORM_DATA).post(
198         String.class, form);
199     return response;
200   }

202   /**
203    * 
204    * http://stackoverflow.com/a/9542781/1497139
205    * 
206    * @param string
207    * @return
208    */
209   public URI convertToValid(String string) {
210     try {
211       String decodedURL = URLDecoder.decode(string, "UTF-8");
212       URL url = new URL(decodedURL);
213       URI uri = new URI(url.getProtocol(), url.getUserInfo(), url.getHost(),
214           url.getPort(), url.getPath(), url.getQuery(), url.getRef());
215       return uri;
216     } catch (Exception ex) {
217       ex.printStackTrace();
218       return null;
219     }
220   }

222   /**
223    * get the given resource for the given path (relative to base url of server
224    * to be started
225    * 
226    * @param path
227    *          - either a relative or absolute path
228    *          for relative paths not starting with "http" the server is started
229    *          and the relative
230    *          path is added to the servers base url
231    * @return
232    * @throws Exception
233    */
234   public WebResource getResource(String path) throws Exception {
235     String url = path;
236     if (!path.startsWith("http")) {
237       url = startServer() + path;
238     }
239     if (debug)
240       System.out.println(url);
241     /*
242      * FIXME - check issue and activate solution
243      * URI uri = convertToValid(url);
244      * if (debug)
245      * System.out.println(uri.toASCIIString());
246      */
247     // path=path.replace("Ã„","%C3%B6");
248     if (rs.getSettings().isSecure()) {
249       client=SSLClientHelper.createClient();
250     } else {
251       client = Client.create();
252     }
253     if (this.user!=null) {
254       client.addFilter(new HTTPBasicAuthFilter(user.getId(),user.getPassword()));
255     }
256     WebResource wrs = client.resource(url);
257     return wrs;
258   }

283   public String getResponseString(ClientResponse response) {
284     if (debug)
285       System.out.println("status: " + response.getStatus());
286     String responseText = response.getEntity(String.class);
287     if (response.getStatus() != 200 && debug) {
288       System.err.println(responseText);
289     }
290     Assert.assertEquals(200, response.getStatus());
291     return responseText;
292   }

356   /**
357    * get a Post response
358    * 
359    * @param contentType
360    * @param path
361    * @param pFormData
362    * @param debug
363    * @return
364    * @throws Exception
365    */
366   public ClientResponse getPostResponse(String contentType,String path,
367       Map<String, String> pFormData, boolean debug) throws Exception {
368     MultivaluedMap<String, String> lFormData = new MultivaluedMapImpl();
369     for (String key : pFormData.keySet()) {
370       lFormData.add(key, pFormData.get(key));
371     }
372     WebResource wrs = getResource(path);
373     ClientResponse response = wrs.type(
374         MediaType.APPLICATION_FORM_URLENCODED_TYPE).accept(contentType).post(ClientResponse.class,
375         lFormData);
376     // do not force response string - further calls to getResponseString will
377     // fail
378     boolean force = false;
379     if (debug && force) {
380       String responseString = this.getResponseString(response);
381       LOGGER.log(Level.INFO, responseString);
382     }
383     return response;
384   }
