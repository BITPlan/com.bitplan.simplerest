40  import com.bitplan.rest.UserManager;
41  import com.bitplan.rest.users.UserImpl;
42  import com.bitplan.rest.users.UserManagerImpl;

44  /**
45   * simple test for Rythm Template based resource
46   * 
47   * @author wf
48   *
49   */
50  public class TestUserManagerResource extends TestHelloServer {
51    protected Logger LOGGER = Logger.getLogger("com.bitplan.rest.test");
52  
53    static UserManager um;
54  
55    /**
56     * get the user manager with a few example users
57     * @return - the usermanager
58     */
59    public static UserManager getUserManager() {
60      if (um == null) {
61        um = UserManagerImpl.getInstance();
62        um.getUsers().clear();
63        um.add(new UserImpl(um, "scott", "Scott", "Bruce",
64            "bruce.scott@tiger.com", "tiger", "CEO", "since 2016-01"));
65        um.add(new UserImpl(um, "scott", "Doe", "John", "john@doe.com",
66            "mightymouse", "Janitor", "since 2017-01"));
67        assertEquals(2, um.getUsers().size());
68      }
69      return um;
70    }
71  
72    @BeforeClass
73    public static void prepareUserManager() {
74      um = getUserManager();
75    }
76  
77    @Test
78    public void testUserManageResource() throws Exception {
79      debug = false;
80      String path = "/hello/users";
81      super.check(path, "Scott");
82      String responseXml = getResponseString("application/xml; charset=UTF-8",
83          path);
84      if (debug) {
85        LOGGER.log(Level.INFO, responseXml);
86      }
87      assertTrue(
88          responseXml.startsWith("<?xml version=\"1.0\" encoding=\"UTF-8\""));
89      responseXml = getResponseString("text/html; charset=UTF-8", path + ".xml");
90      assertTrue(
91          responseXml.startsWith("<?xml version=\"1.0\" encoding=\"UTF-8\""));
92      String responseJson = getResponseString(
93          MediaType.APPLICATION_JSON + "; charset=UTF-8", path);
94      if (debug) {
95        LOGGER.log(Level.INFO, responseJson);
96      }
97      assertTrue("Json", responseJson.contains("\"users\":{\"User\""));
98      responseJson = getResponseString("text/html; charset=UTF-8",
99          path + ".json");
100     assertTrue("Json", responseJson.contains("\"users\":{\"User\""));
101   }
